    其他参数
    autoWidth: false / true //轮播区域是否自适应宽度（默认false）
    loop: false / true //是否无视差循环
    pager:".js_pager" //分页容器class或id
    speed:1000  //自动轮播间隙时间(毫秒)
    moveSpeed:500 //轮播切换的移动速度
    pagerShow: false / true //分页是否显示索引内容
    pagerHover:true/false //分页是否hover切换
    prevFn:function(i){} //移动到上一个的回调函数 参数i:当前轮播索引
    nextFn:function(i){} //移动到下一个的回调函数 参数i:当前轮播索引
    playFn:function(p){} //切换时的回调函数不分上下（在prevFn/nextFn后执行） 参数json：p.i=播放的页数索引0开始 / p.next=按钮next / p.prev=按钮prev
    windowResize：true/false //是否支持浏览器改变大小后重置轮播（默认true）
    moveOne:ture/false //滑动一次只切换个轮播 (默认false)
    moveTouch:true/false //根据滑动幅度切换轮播个数 (默认ture)
    directionVertical : true/false  //定义纵向 (默认false)

    其他属性例如“bannerSlider”
    bannerSlider.prevBtn //轮播按钮（上一个）
    bannerSlider.nextBtn //轮播按钮（下一个）
    bannerSlider.clickI //当前点击的轮播块索引
    startFn:function(){} //初始化完毕回调

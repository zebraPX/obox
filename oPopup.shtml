    //弹层初始化
    var myPopup=$(".js_popupbtn").oPopup({
      confirmFn:function(p){
        alert(p.selfEle.html());
      }
    });
    //弹层内部内部要初始化高度试用下面方法
    //mypopup.init();

    //其他参数
    confirm:弹层内的确认按钮标识(默认".js_confirm")。
    confirmFn:function(p){}确认后要执行的方法内置josn参数p下面是参数介绍
         p.selfEle=触发弹层的按钮,
         p.targetEle=弹层本身。
    closeFn:function(){}关闭弹层时要执行的方法(默认：null)。
    openFn:function(){}打开弹层时执行的方法(默认：null)。
    //其他方法
    myPopup.open();打开弹层
    myPopup.close();关闭弹层
	shadeClose:false 点击遮照是否可以关闭弹层 false/true（默认：false）

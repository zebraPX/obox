    //弹层初始化（可管理本页面所有弹层，初始化一次就可以）
    var myPopup=$(".o_body").oPopupFn();

    //弹出弹层
    myPopup.open({
        target:".js_tanceng2"//要弹出的弹层
    })

    //例子（第一次点击弹出第一个弹层，第二次点击弹出第二个弹层）
     $(".js_popupbtn").oToggle(function(){
        myPopup.open({
         target:".js_tanceng1"
        })
    },function(){
        myPopup.open({
         target:".js_tanceng2"
        })
    })

    //open方法其他参数
    closeFn:function(){}关闭弹层时要执行的方法(默认：null)。
    //对象其他方法
    myPopup.open();打开弹层
    myPopup.close();关闭弹层

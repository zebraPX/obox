    <div class="js_listPager"></div>
    
    // 2. 创建分页对象 
    var pp = $(".js_listPager").oPager({
                                        btnLength:3, //显示几个页码
                                        goPageFn:function(s){
                                            loadList(s)
                                        }
                                    });

    //重要参数
    （必填）goPageFn:function(i){} 用于绑定页面跳转事件参数i为跳转的页码

    //其他参数
    prevText:上一页按钮显示的文字 默认:"上一页"
    nextText:下一页按钮显示的文字 默认:"下一页"
    btnLength:显示的页码个数 默认：3
    btnlength_sm:sm下显示的页面个数 默认：3
    btnlength_md:md下显示的页面个数 默认：3
    btnlength_xs:xs下显示的页面个数 默认：3
    showJump:（true/false）是否显示跳转条， 默认：true
    jumpAddClass:跳转条上要添加的样式，可以用来控制在不同节点下的显示隐藏，默认："o_sm-hide o_xs-hide"
    jumpText:跳转条上的辅助文字1 默认：“跳转到”
    jumpText2:跳转条上的辅助文字2 默认：“页”
    showLast:(true/false)是否展示最后页码 默认：“true”

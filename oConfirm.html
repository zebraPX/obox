    //confirm初始化(初始化一次就可以)
    var myConfirm=$(".o_body").oConfirm();

    //调用弹出删除一条信息
    $(".js_del").on("click",function(){

       $(".js_del").on("click",function(){
            myConfirm.open({
                info : "你确定要删除这条信息么？", //提示信息
                ele : $(this), //传递对象
                callbackFn : function(ele){ //确认后执行方法
                    ele.parent().remove();
                    },
                addClass:"red" //给弹层添加其他class
                }); 
    }); 
        
    });

    //调用弹出删除全部信息
    $(".js_delall").on("click",function(){
        myConfirm.open({
            info : "你确定要删除全部信息么？",
            ele : $(".list"),
            callbackFn : function(ele){
                ele.find("li").remove();
                }
            }); 
    });

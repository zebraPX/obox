oHoverMove / hover移动事件

鼠标经过区域

    <div class="js_obj" >触摸区域</div>

 $(".js_obj").oHoverMove({
        mouseStart:function(p){
           $(".js_text").html($(".js_text").html()+" hover ");
           //console.log(p);
        },
        mouseMove:function(p){
           $(".js_text").html($(".js_text").html()+" move ");
           //console.log(p);
        },
        mouseLeave:function(p){
             $(".js_text").html($(".js_text").html()+" leave ");
             //console.log(p);
        },
        mouseStopFn:function(p){
             $(".js_text").html($(".js_text").html()+" 停止了 ");
             console.log(p);
        },
        stopTime:3000
    })

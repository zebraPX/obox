oTouch / 触摸事件

触摸区域

    <div class="js_obj" >触摸区域</div>

 $(".js_obj").oTouch({
        touchStart:function(p){//点击
            $(".js_text").html($(".js_text").html()+"<span style='color:red'>开始 "+p.self.html()+"</span>");
            console.log(p.self.attr("class"))
        },
        touchMove:function(p){//移动中
            $(".js_text").html($(".js_text").html()+" move ");
            console.log(p)
        },
        touchEnd:function(p){//touch结束
            $(".js_text").html($(".js_text").html()+"<span style='color:green' > touchend </span>");
            console.log(p)
        }
    })
    //查看log了解参数

    其他参数
    touchMoveH:function(p){...}//只监听横向滑动
    touchMoveV:function(p){...}//只监听纵向滑动
    touchLeft:function(p){...}//同touchEnd
    touchRight:function(p){...}//同touchEnd
    touchUp:function(p){...}//同touchEnd
    touchDown:function(p){...}//同touchEnd

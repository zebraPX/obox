     $(".js_door").oToggle(function(s){
        $(".js_text").html($(".js_text").html()+" open ");
    },function(s){
        $(".js_text").html($(".js_text").html()+" close ");
    });

    $(".js_close").on("click",function(){
        $(".js_door").attr("auto","0");
    })

    $(".js_doorold").toggle(function(){
        $(".js_text").html($(".js_text").html()+" open ");
    },function(){
        $(".js_text").html($(".js_text").html()+" close ");
    });

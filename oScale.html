oScale / 比例大小

1:1
2:1
2:3

适用于固定比例的元素，长或宽必须有一项有明确有值

     <!--css-->
         .box{width:100px;}
        @media (min-width:1200px){
            .box{width:200px;}
        }

   
    <!--使用方法1 js传参-->
    <div class="js_box box" >1:1</div><!--没填参数 默认1：1-->
    <!--js-->
    var box1=$(".js_box").oScale()//创建方法
    box1.init();//激活方法

    <div class="js_box2 box" >2:1</div><!--js传参 2：1-->
    <!--js-->
    var box2=$(".js_box2").oScale({
        w:2,
        h:1
        })//js传参
    box2.init();


    <!--使用方法2 html传参-->
    <div class="js_box box" w="2" h="3" base="width">2:3</div>
    <!--js-->
    var box1=$(".js_box").oScale()//创建方法
    box1.init();//激活方法

加载区域

开始加载 加载完毕 
        <div class="showbox">加载区域</div>
        <a href="javascript:;" class="o_bgbtn1 o_btn_df-df js_loading"><span>开始加载</span></a>
        <a href="javascript:;" class="o_bgbtnbase o_btn_df-df js_loaded"><span>加载完毕</span></a>
        </div>
    
        $(function(){
            //初始化loading对象
            var loading=$(".showbox").oLoading();
    
            $(".js_loading").on("click",function(){
                //开始加载
                loading.start();
            });
    
            $(".js_loaded").on("click",function(){
                //加载完成
                loading.end();
            });
    
        });
